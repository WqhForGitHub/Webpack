                    ### 5.1 分离样式文件

3.  mini-css-extract-plugin

   ```javascript
   const MiniCssExtractPlugin = require('mini-css-extract-plugin');
   
   module.exports = {
       entry: './app.js',
       output: {
           filename: '[name].js'
       },
       mode: 'development',
       module: {
           rules: [
               {
                   test: /\.css$/,
                   use: [
                       {
                           loader: MiniCssExtractPlugin.loader,
                           options: {
                               publicPath: '../'
                           }
                       },
                       'css-loader'
                   ]
               }
           ]
       },
       plugins: [
           new MiniCssExtractPlugin({
               filename: '[name].css',
               chunkFilename: '[id].css'
           })
       ]
   }
   ```
